# Seasonal Leaderboards: Fresh Starts and New Opportunities Every Quarter

Perpetual leaderboards have a fatal flaw: early adopters dominate forever, newcomers have no realistic path to the top, and everyone else gives up. Seasonal leaderboards solve this with regular resets that give everyone a fresh start.

## The Perpetual Leaderboard Problem

### Why Always-On Leaderboards Fail

```typescript
const perpetualLeaderboard = {
  year1: {
    top10: ['Early adopters'],
    everyone_else: 'No chance of catching up'
  },

  year2: {
    top10: ['Same early adopters'],
    everyone_else: 'Still no chance',
    newFounders: 'Why even try?'
  },

  year3: {
    engagement: 'Only top players care',
    newSignups: 'Ignore leaderboard',
    motivation: 'System has failed'
  }
};
```

Problems:
- **Entrenched leaders**: Top spots never change
- **Newcomer discouragement**: "I'm 10,000 points behind, why bother?"
- **Reduced competition**: Most founders stop competing
- **Stale rankings**: No excitement or drama

## Season Mechanics

### Season Start

Everyone begins at zero:

```typescript

export const resetSeasonalLeaderboards = async () => {
  const founders = await getAllFounders();

  for (const founder of founders) {
    await updateFounder(founder.id, {
      seasonalXP: 0,
      seasonalRank: null,
      seasonalAchievements: [],
      seasonStartLevel: founder.level, // Track for improvement rewards
      seasonStartMRR: founder.metrics.mrr
    });
  }

  // Create fresh leaderboards
  await createSeasonLeaderboards();
};
```

### During Season

Real-time competition:

```typescript
export const updateSeasonalLeaderboard = agenticFunction({
  schedule: 'every 15 minutes',
  async execute() {
    const season = await getCurrentSeason();
    const founders = await getActiveFounders();

    // Calculate seasonal scores
    const scores = await Promise.all(
      founders.map(async f => ({
        founderId: f.id,
        seasonalXP: f.seasonalXP,
        seasonalScore: await calculateSeasonalScore(f, season)
      }))
    );

    // Update rankings
    scores.sort((a, b) => b.seasonalScore - a.seasonalScore);

    for (let i = 0; i < scores.length; i++) {
      const entry = scores[i];
      const previousRank = await getSeasonalRank(entry.founderId);
      const newRank = i + 1;

      await updateSeasonalRank(entry.founderId, newRank);

      // Notify of significant rank changes
      if (previousRank && Math.abs(newRank - previousRank) >= 5) {
        await notifyRankChange({
          founderId: entry.founderId,
          oldRank: previousRank,
          newRank,
          direction: newRank < previousRank ? 'up' : 'down'
        });
      }
    }

    await saveSeasonalLeaderboard(season.id, scores);
  }
});
```

### Season End

Celebrate and reward:

```typescript
export const endSeason = agenticFunction({
  async execute() {
    const season = await getCurrentSeason();
    const leaderboard = await getSeasonalLeaderboard(season.id);

    // Award seasonal rewards
    await awardSeasonalRewards({
      season,
      leaderboard,
      rewards: season.rewards
    });

    // Generate season recap
    for (const entry of leaderboard) {
      const recap = await generateSeasonRecap(entry.founderId, season);
      await sendSeasonRecap(entry.founderId, recap);
    }

    // Announce winners
    await announceSeasonWinners(season, leaderboard.slice(0, 10));

    // Archive season data
    await archiveSeason(season);

    // Prepare next season
    await prepareNextSeason();
  }
});

async function generateSeasonRecap(founderId: string, season: Season) {
  const founder = await getFounder(founderId);
  const seasonData = await getSeasonData(founderId, season.id);

  return {
    rank: seasonData.finalRank,
    totalFounders: seasonData.totalParticipants,
    percentile: (seasonData.finalRank / seasonData.totalParticipants) * 100,

    achievements: {
      seasonalXP: seasonData.totalXP,
      achievementsUnlocked: seasonData.achievements.length,
      questsCompleted: seasonData.quests.length,
      streakDays: seasonData.activeDays
    },

    growth: {
      levelGain: founder.level - seasonData.startLevel,
      mrrGrowth: founder.metrics.mrr - seasonData.startMRR,
      customersAdded: seasonData.customersAdded,
      deploymentsCount: seasonData.deployments
    },

    highlights: await selectSeasonHighlights(seasonData),

    comparison: {
      vsLastSeason: await compareToLastSeason(founderId, season),
      vsCohort: await compareToSeasonal Cohort(founderId, season)
    },

    rewards: seasonData.rewardsEarned
  };
}
```

export const trackComebackStory = codeFunction({
  input: z.object({
    founderId: z.string(),
    seasonId: z.string()
  }),
  async execute({ founderId, seasonId }) {
    const lastSeason = await getLastSeason(seasonId);
    const lastRank = await getSeasonRank(founderId, lastSeason.id);
    const currentRank = await getCurrentSeasonRank(founderId);

    if (lastRank > 100 && currentRank <= 10) {
      // Dramatic comeback!
      await unlockAchievement(founderId, 'comeback-kid');

      await featuredStory({
        founderId,
        title: 'The Comeback Story of the Season',
        data: {
          lastSeason: { rank: lastRank },
          thisSeason: { rank: currentRank },
          improvement: lastRank - currentRank
        }
      });
    }

    return { comeback: currentRank < lastRank };
  }
});
